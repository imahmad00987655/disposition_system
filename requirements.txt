# ============================================
# System Requirements
# ============================================
# Node.js: 20.x or higher
# Docker: 20.x or higher (for containerized deployment)
# Docker Compose: 2.x or higher (optional)

# ============================================
# Backend Requirements
# ============================================
# PHP: 7.4+ (for PHP API endpoints)
# Apache/Nginx: For serving PHP files
# SQL Server: Access to database at 192.168.1.6:1433
# MySQL: Access to database at 192.168.1.209

# ============================================
# Network Requirements
# ============================================
# Port 6126: Next.js application (configurable)
# Port 80/443: PHP backend (Apache/Nginx)
# Port 6004: Complaints API
# Port 5125: Orders API
# Port 1433: SQL Server database
# Port 3306: MySQL database

# ============================================
# Environment Variables Required
# ============================================
# NEXT_PUBLIC_API_BASE_URL: Base URL for PHP API endpoints
# NEXT_PUBLIC_COMPLAINTS_API: Complaints API endpoint
# NEXT_PUBLIC_ORDERS_API: Orders API endpoint
# NODE_ENV: production/development
# PORT: Application port (default: 6126)

# ============================================
# Installation Commands
# ============================================
# Local Development:
#   npm install
#   npm run dev

# Production Build:
#   npm ci --only=production
#   npm run build
#   npm start

# Docker Deployment:
#   docker-compose up -d --build
#   OR
#   docker build -t disposition-system .
#   docker run -p 6126:6126 disposition-system
